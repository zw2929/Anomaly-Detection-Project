Section 3.1: Using PCA for Feature1-9 ¶

PCA which is Principal Component Analysis is a method that reduces the dimensionality of a dataset by transforming correlated features into a smaller set of uncorrelated components, making it easier to spot patterns and identify outliers in the data. Apply the PCA model to calculate the outlier score for each hospital based on its distance from the main cluster in this simplified space. The contamination = 0.05 here means 5% of the data points will be considered as outliers.¶

Here, I divided my features into two groups: features 1-9 and features 10-15, because I noticed that these two groups have different scopes. Specifically, features 1-9 focus on comparing hospital payment and discharge ratios relative to state or city benchmarks, while features 10-15 concentrate on similar ratios but specifically for Medicare payments and payments per discharge. This distinction allows for a more targeted analysis of general payment patterns versus Medicare-specific and discharge-specific payment behaviors.¶

# Create feature list
feature_list = ['F1', 'F2', 'F3', 'F4', 'F5', 'F6', 'F7', 'F8', 'F9']

def descriptive_stat_threshold(X, scores, threshold):
    normal_mask = scores <= threshold
    anomaly_mask = scores > threshold

    normal_stats = X[normal_mask].describe()
    anomaly_stats = X[anomaly_mask].describe()

    summary_df = pd.DataFrame({
        'Group': ['Normal', 'Outlier'],
        'Count': [normal_mask.sum(), anomaly_mask.sum()],
        'Count %': [normal_mask.mean() * 100, anomaly_mask.mean() * 100]
    })
    
    for i, feature in enumerate(feature_list, 1):
        summary_df[f'Feature_{i}'] = [
            X[normal_mask][feature].mean(),
            X[anomaly_mask][feature].mean()
        ]
    
    summary_df['Anomaly_Score'] = [
        scores[normal_mask].mean(),
        scores[anomaly_mask].mean()
    ]
    
    return summary_df

def knn_stat_summary(X, pred_scores, threshold):
    # Create DataFrame and set columns
    df = pd.DataFrame(X)
    df['Anomaly_Score'] = pred_scores
    df['Group'] = np.where(df['Anomaly_Score'] < threshold, 'Normal', 'Outlier')

    # Count and percentage of each group
    cnt = df.groupby('Group')['Anomaly_Score'].count().reset_index().rename(columns={'Anomaly_Score': 'Count'})
    cnt['Count %'] = (cnt['Count'] / cnt['Count'].sum()) * 100 

    # Calculate mean statistics
    stat = df.groupby('Group').mean().round(2).reset_index() 

    # Merge count and average statistics
    stat = cnt.merge(stat, left_on='Group', right_on='Group') 
    return stat

def descriptive_stat_threshold(X, scores, threshold):
    normal_mask = scores <= threshold
    anomaly_mask = scores > threshold

    normal_stats = X[normal_mask].describe()
    anomaly_stats = X[anomaly_mask].describe()

    summary_df = pd.DataFrame({
        'Group': ['Normal', 'Outlier'],
        'Count': [normal_mask.sum(), anomaly_mask.sum()],
        'Count %': [normal_mask.mean() * 100, anomaly_mask.mean() * 100]
    })
    
    for i, feature in enumerate(feature_list, 1):
        summary_df[f'Feature_{i}'] = [
            X[normal_mask][feature].mean(),
            X[anomaly_mask][feature].mean()
        ]
    
    summary_df['Anomaly_Score'] = [
        scores[normal_mask].mean(),
        scores[anomaly_mask].mean()
    ]
    
    return summary_df

def knn_stat_summary(X, pred_scores, threshold):
    df = pd.DataFrame(X)
    df['Anomaly_Score'] = pred_scores
    df['Group'] = np.where(df['Anomaly_Score'] < threshold, 'Normal', 'Outlier')

    cnt = df.groupby('Group')['Anomaly_Score'].count().reset_index().rename(columns={'Anomaly_Score': 'Count'})
    cnt['Count %'] = (cnt['Count'] / cnt['Count'].sum()) * 100 

    stat = df.groupby('Group').mean().round(2).reset_index() 

    stat = cnt.merge(stat, left_on='Group', right_on='Group') 
    return stat
